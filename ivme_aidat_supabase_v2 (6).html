Supabase SQL Güncelleme Kodu
Bu programı ilk defa kuruyorsanız aşağıdaki SQL kodunu Supabase → SQL Editor'da çalıştırın. Daha önce eski tabloları kurduysanız önce onları silip bu yeni kodu çalıştırın. 
-- Eski tabloları sil (varsa)
drop table if exists sms_log;
drop table if exists aidatlar;
drop table if exists ogrenciler;

-- Öğrenciler tablosu
create table ogrenciler (
  id bigint generated always as identity primary key,
  ad_soyad text not null,
  sinif text,
  tc_no text,
  anne_tel text,
  baba_tel text,
  telefon text,
  toplam_ucret numeric default 0,
  pesinat numeric default 0,
  taksit_sayisi int default 1,
  taksit_tutari numeric default 0,
  odeme_periyodu int default 1,
  odeme_sekli text default 'Elden',
  aciklama text,
  created_at timestamptz default now()
);

-- Taksitler tablosu
create table aidatlar (
  id bigint generated always as identity primary key,
  ogrenci_id bigint references ogrenciler(id) on delete cascade,
  taksit_no int default 1,
  ay text,
  tutar numeric default 0,
  son_odeme_tarihi date,
  odeme_tarihi date,
  odeme_sekli text,
  durum text default 'ÖDENMEDI',
  sms_gonderildi boolean default false,
  sms_tarihi timestamptz,
  created_at timestamptz default now()
);

-- SMS geçmişi
create table sms_log (
  id bigint generated always as identity primary key,
  ogrenci_id bigint,
  telefon text,
  mesaj text,
  durum text,
  api_yanit text,
  created_at timestamptz default now()
);
